### 试一下

```objc
void MergeSort(int arr1[], int arr2[],int len1,int len2)
{
	if(arr1 == NULL || arr2 == NULL || len1 < 0 || len2 < 0)
		return ;
	int i,j,k;
	i = len1 - 1;
	j = len2 - 1;
	k = len1 + len2 - 1;
	while(i >= 0 && j >= 0)
	{
		if(arr1[i] > arr2[j])
			arr1[k--] = arr1[i--];
		else
			arr1[k--] = arr2[j--];
	}
	while(i >= 0)
	{
		arr1[k--] = arr1[i--];
	}
	while(j >= 0)
	{
		arr1[k--] = arr2[j--];
	}
}
```
