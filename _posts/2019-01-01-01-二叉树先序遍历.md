### 二叉树先序遍历


```objc
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> stval;
        if(root==NULL)
            return stval;
        TreeNode* p=root;
        stack<TreeNode*> s;
        while(!s.empty()||p!=NULL)
        {
            while(p!=NULL)//先可这
            {
                stval.push_back(p->val);
                s.push(p);
                p=p->left;
            }
            if(!s.empty())
            {
                p=s.top();
                s.pop();
                p=p->right;
            }
        }
        return stval;
        
    }
};
```
