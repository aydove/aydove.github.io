### 试一下

```objc
class Solution
{
public:
    vector<int> printListFromTailToHead(struct ListNode* head)
    {
        ListNode *node = head;
        stack<int> st;
        int count = 0;
        while(node != NULL)
        {
            st.push(node->val);
            count++;
            node = node->next;
        }
        //  为了效率我们静态vector开辟空间
        vector<int> res(count);
        for(int i = 0; i < count && st.empty( ) != true; i++)
        {
            res[i] = st.top( );
            st.pop( );
        }
        return res;
    }
};
```
