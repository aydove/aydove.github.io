### 二进制中1的个数
输入一个整数，输出该数二进制表示中1的个数。其中负数用补码表示。

```objc
#include <iostream>
using namespace std;
class Solution
{
public:
    int NumberOf1(int n)
    {
        int count = 0;
// 不要忘记赋初值0啊啊啊！！！

        while(n)
        {
            ++count;  
            //先不做&操作就加1，所以最后一次&操作后n为0没加1
            
            n = (n - 1) & n;
        }
        return count;
    }
};
```

下面这种如果是32位数就需要比较32次，注意这里是指32位64位这样的数

```objc
class Solution
{
public:
    int NumberOf1(int n)
    {
        int count = 0;
// 不要忘记赋初值0啊啊啊！！！

        unsigned int flag = 1;
        while(flag != 0)
        {
            if((n & flag) != 0)
// &‘ 与或非’这类运算优先级最低，！优先，所以需要+（）

//不想加可以直接写if(n&flag)


            {
                count++;
            }
            flag <<= 1;
        }
        return count;
    }
};
```
