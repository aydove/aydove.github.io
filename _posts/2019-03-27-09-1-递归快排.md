#include <iostream>  

#include <vector>  

using namespace std;
int Partition(vector<int> &A,int p,int r){
    int x=A[r];
    int small=p;
    int j;
    for(j=p;j<r;j++)
    {
        if(A[j]<x)
        {
            swap(A[small],A[j]);
            small++;
        }
    }
    swap(A[small],A[r]);
    return small;
}
 
void QuickSort(vector<int> &A,int p,int r){
    if(p<r)
    {
        int q=Partition(A,p,r);
        QuickSort(A,p,q-1);
        QuickSort(A,q+1,r);
    }
}
int main(){
    int n;
    cin>>n;
    vector<int> A(n);
    for(int i=0;i<n;++i)
        cin>>A[i];
    QuickSort(A,0,n-1);
    for(int i=0;i<n;++i){
        cout<<A[i];
        if(i!=n-1)
            cout<<" ";
    }
}
