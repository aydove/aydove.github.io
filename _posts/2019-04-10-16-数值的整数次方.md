### 数值的整数次方

**0的正数次方为0**

**0的负数次方为没有意义**

**0的0次方为1或0都可以，这里认为是1**

```objc
class Solution {
public:
    bool equal(double num1, double num2)
{
    if ((num1 - num2 > -0.0000001) && (num1 - num2 < 0.0000001))
        return true;
    else
        return false;
}

    double Power(double base, int exponent) 
{

// 5种情况  

//  :0的负数次方为没有意义  

        if(equal(base,0.0)&&exponent<0)
            return false;  
//  0的正数次方为0  

        else if(equal(base,0.0)&&exponent>0)
            return 0;  
//   正数次方  

        else if(exponent>0)
        {
            if(exponent==1)
                return base;
            double result=Power(base,exponent>>1);
            result*=result; //偶数结果
            
            if(exponent&0x1==1)
                result*=base; //奇数结果
                
            return result;        }
                
//  0次方
         
        else if (exponent==0)
        {
            return 1;
        }

//  负数次方
        
        else
        {
            return 1/Power(base,0-exponent);
        }
    }
};
```
