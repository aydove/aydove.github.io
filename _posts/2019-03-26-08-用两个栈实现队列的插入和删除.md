### 试一下

'''
class Solution
{
public:
//添加
    void push(int node) {
        stack1.push(node);
    }
//删除
    int pop() {
        if(stack2.size()<=0)
        {
            while(stack1.size()>0)
            {
                int i=stack1.top();
                stack1.pop();
                stack2.push(i);
            }
        }
        if (stack2.size()==0)
        {
            return false;
        }
        int head = stack2.top();
        stack2.pop();
        return head;
    }
private:
    stack<int> stack1;
    stack<int> stack2;
};
'''
