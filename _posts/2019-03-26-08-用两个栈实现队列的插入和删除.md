### 试一下

用两个栈来实现一个队列，完成队列的Push和Pop操作。 队列中的元素为int类型。

```objc
class Solution
{
public:  
//添加

    void push(int node) {
        stack1.push(node);
    }  
//删除

    int pop() {  
        //1先装入stack2中
        
        if(stack2.size()<=0)
        {
            while(stack1.size()>0)
            {
                int i=stack1.top();
                stack1.pop();
                stack2.push(i);
            }
        }  
        //2检查一下边界
        
        if (stack2.size()==0)
        {
            return false;
        }  
        //3再删除，别忘了return
        
        int head = stack2.top();
        stack2.pop();
        return head;
    }
private:
    stack<int> stack1;
    stack<int> stack2;
};
```
